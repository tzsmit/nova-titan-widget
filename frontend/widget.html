Embeddable AI-Powered Sports Predictions

How to Embed

Basic Embedding:

<script src="https://your-domain.com/widget.js" data-api-url="https://your-api.com" data-theme="nova-titan"></script> <div id="nova-titan-widget"></div>

WordPress Integration:

// Add to your theme's functions.php function add_nova_titan_widget() { wp_enqueue_script('nova-titan-widget', 'https://your-domain.com/widget.js', array(), '1.0.0', true); } add_action('wp_enqueue_scripts', 'add_nova_titan_widget'); // Use shortcode: [nova_titan_widget]

React Integration:

import { NovaTitanWidget } from '@nova-titan/react-widget'; function App() { return ( <NovaTitanWidget apiUrl="https://your-api.com" theme="nova-titan" leagues={['NBA', 'NFL']} enablePredictions={true} /> ); }
